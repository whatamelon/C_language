/*

/*
변수 초기화 -> ex) int age =20; or int age; age=20;
*/

/*
변수 동시 선언 -> ex) int a =10; int b=20;
*/

//변수 선언하고 초기화하기
#include <stdio.h>

int main(void)
{
	int num1 = 10, num2 = 5;

	printf("num1의 초깃값은 %d이다. \n", num1);
	printf("num2의 초깃값은 %d이다. \n", num2);

	num1 = 30;
	num2 = 20;

	printf("num1의 새로운 값은 %d이다. \n", num1);
	printf("num2의 새로운 값은 %d이다. \n", num2);

	return 0;
}
